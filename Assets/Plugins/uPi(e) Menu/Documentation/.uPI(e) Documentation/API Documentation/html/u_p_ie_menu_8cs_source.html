483  public Vector2 GetIndicatorPosition(Vector2 dir, out Vector2 resultPosition)
484  {
485  if (IndicatorGraphic == null)
486  {
487  resultPosition = new Vector2();
488  return Vector2.zero;
489  }
490 
491  var myRectTrans = (RectTransform)IndicatorGraphic.transform.parent;
492  if (myRectTrans == null)
493  {
494  myRectTrans = (RectTransform)transform;
495  }
496 
497  Vector2 constrainedDir = Vector2.zero;
498  if (ConstrainIndicatorPosition && CircleSize < 360)
499  {
500  var startDir = GetStartDirection();
501  var endDir = GetEndDirection();
502  var angleFromStart = GetSignedAngle(startDir, dir);
503  var angleFromEnd = GetSignedAngle(endDir, dir);
504  if (angleFromStart < 0)
505  {
506  angleFromStart += 360;
507  }
508 
509  var deadAngle = 360 - CircleSize;
510  var midDir = Quaternion.Euler(0, 0, deadAngle / 2) * startDir;
511  var angleToMid = GetSignedAngle(midDir, dir);
512  if (angleFromStart > CircleSize && angleToMid > 0)
513  {
514  constrainedDir = dir = startDir;
515  }
516  else if (angleFromEnd > 0 && angleToMid < 0)
517  {
518  constrainedDir = dir = endDir;
519  }
520  }
521 
522  Vector2 elipticRadius = new Vector2(myRectTrans.sizeDelta.x * myRectTrans.pivot.x, myRectTrans.sizeDelta.y * myRectTrans.pivot.y);
523  resultPosition = Vector2.Scale(dir, elipticRadius);
524  return constrainedDir;
525  }

583  private float GetAngle(Vector2 dir, float degOffset = 0, bool zeroTo360 = false)
584  {
585  var offsettedDir = GetOffsettedDir(dir, degOffset);
586  var atan = Mathf.Atan2(offsettedDir.y, offsettedDir.x);
587  var degAngle = atan * Mathf.Rad2Deg;
588  if(zeroTo360)
589  {
590  degAngle -= 180;
591  degAngle = Mathf.Abs(degAngle);
592  }
593 
594  return degAngle;
595  }

928  private void Update()
929  {
930  if (!EnableSelecting && !ControlWithGamepad) return;
931 
932  PollInput();
933  currentDirection = GetDirection();
934 
935  // Set default with gamepad
936  if (ControlWithGamepad)
937  {
938  float curDirMag = CustomInput.sqrMagnitude;
939  if (curDirMag < ControllerDeadzone)
940  {
941  if (DefaultSelected != null)
942  {
943  if (CurrentlyActiveOption != DefaultSelected)
944  {
945  CurrentlyActiveOption = DefaultSelected;
946  DefaultSelected.Select();
947  prevSelectedPieceId = -1;
948  SelectedPieceId = -1;
949 
950  if (IndicatorGraphic != null)
951  {
952  IndicatorGraphic.enabled = false;
953  }
954  }
955  }
956 
957  return;
958  }
959  else
960  {
961  if (IndicatorGraphic != null)
962  {
963  if (!IndicatorGraphic.enabled)
964  {
965  IndicatorGraphic.enabled = true;
966  }
967  }
968  }
969  }
970 
971  var angle = GetAngle(currentDirection, StartDegOffset, true);
972  SelectedPieceId = GetPieceByAngle(angle);
973  if (SelectedPieceId != prevSelectedPieceId)
974  {
975  if (SelectedPieceId >= 0)
976  {
977  CurrentlyActiveOption = SelectRelatedOption();
978  }
979  else
980  {
981  if (DeselectOptionIfOutsideBorders && CircleSize < 360f)
982  {
983  Deselect();
984  }
985  }
986  }
987 
988  prevSelectedPieceId = SelectedPieceId;
989  if (ConfirmButtonDown)
990  {
991  if (CurrentlyActiveOption != null)
992  {
993  EventSystem.current.SetSelectedGameObject(null);
994  var pointer = new PointerEventData(EventSystem.current);
995  ExecuteEvents.Execute(CurrentlyActiveOption.gameObject, pointer, ExecuteEvents.submitHandler);
996  }
997 
998  ConfirmButtonDown = false;
999  }
1000 
1001  // Indicator draw stuff
1002  if (IndicatorGraphic == null) return;
1003 
1004  Vector2 resultPosition;
1005  Vector2 constrainedDir = GetIndicatorPosition(currentDirection, out resultPosition);
1006  if (ApplyIndicatorRotation)
1007  {
1008  if (constrainedDir != Vector2.zero)
1009  {
1010  indicatorGraphic.transform.right = constrainedDir;
1011  }
1012  else
1013  {
1014  IndicatorGraphic.transform.right = currentDirection;
1015  }
1016  }
1017 
1018  IndicatorGraphic.transform.localPosition = resultPosition;
1019  }

1037  private void DrawGizmos()
1038  {
1039  Vector3 startDir = GetStartDirection();
1040  var rectTrans = (RectTransform)transform;
1041  var maxSize = Mathf.Max(rectTrans.sizeDelta.x, rectTrans.sizeDelta.y);
1042  var deltaSize = new Vector3(rectTrans.pivot.x * maxSize, rectTrans.pivot.y * maxSize, 0);
1043 
1044  // Draw parts/ pieces
1045  var anglePerPiece = CircleSize / menuOptions.Count;
1046  Gizmos.color = Color.black;
1047  for (int i = 0; i < menuOptions.Count; i++)
1048  {
1049  var pieceStartDir = Quaternion.Euler(0, 0, -anglePerPiece * i) * startDir;
1050  Gizmos.DrawRay(transform.position, transform.TransformDirection(Vector2.Scale(pieceStartDir, deltaSize)) * BoundaryLength);
1051  }
1052 
1053  // draw end line
1054  Gizmos.color = Color.yellow;
1055  Vector3 endDir = GetEndDirection(startDir);
1056  Gizmos.DrawRay(transform.position, transform.TransformDirection(Vector2.Scale(endDir, deltaSize)) * BoundaryLength);
1057 
1058  // Draw start line. At last, so it is drawn above all other stuff
1059  Gizmos.color = Color.green;
1060  Gizmos.DrawRay(transform.position, transform.TransformDirection(Vector2.Scale(startDir, deltaSize)) * BoundaryLength);
1061  }
