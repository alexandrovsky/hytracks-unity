400  private void RemoveMenuOption(uPIeMenu targetScript, Selectable slct)
401  {
402  if (slct == null) return;
403 
404  var index = targetScript.MenuOptions.IndexOf(slct);
405 
406  // Somehow DeleteArrayElementAtIndex does not actually delete
407  // the element from the array (like List<T>().RemoveAt(int) would do)
408  // but just sets the element at the selected index to null
409  menuOptions.DeleteArrayElementAtIndex(index);
410 
411  // apply modified now, so und also connects the option
412  // back to the selectable we are going to delete
413  serializedObject.ApplyModifiedProperties();
414 
415  // to actually delete the array element we need to call the delete function again
416  menuOptions.DeleteArrayElementAtIndex(index);
417  Undo.DestroyObjectImmediate(slct.gameObject);
418  }
